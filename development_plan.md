### 6.2 Backend Services (API Endpoints)

#### **Authentication Service**
```
POST   /auth/email              # Sign in with email/password
POST   /auth/google             # Sign in with Google OAuth
POST   /auth/signup             # Register new user with email
POST   /auth/refresh            # Refresh token
POST   /auth/forgot-password    # Request password reset
POST   /auth/reset-password     # Reset password with token
DELETE /auth/logout             # Logout
```

#### **Users Service**
```
GET    /users/me                # Get current user profile
PATCH  /users/me                # Update profile
POST   /users/me/interests      # Set interests
GET    /users/:id               # Get public profile
GET    /users/:id/badges        # Get user badges
DELETE /users/me                # Delete account
```

#### **Clubs Service**
```
GET    /clubs                   # List all clubs
GET    /clubs/:id               # Get club details
POST   /clubs/:id/join          # Join club
DELETE /clubs/:id/leave         # Leave club
GET    /clubs/:id/members       # List club members
GET    /clubs/:id/events        # List club events
```

#### **Events Service**
```
GET    /events                  # List events (with filters)
GET    /events/:id              # Get event details
POST   /events                  # Create event
PATCH  /events/:id              # Update event
DELETE /events/:id              # Cancel event
GET    /events/:id/attendees    # List attendees
POST   /events/:id/rsvp         # RSVP to event
DELETE /events/:id/rsvp         # Cancel RSVP
POST   /events/:id/checkin      # Check-in to event
GET    /events/:id/chat         # Get event chat
POST   /events/:id/chat         # Send message
```

#### **Bookings Service**
```
POST   /bookings                # Create booking (paid event)
GET    /bookings/:id            # Get booking details
POST   /bookings/:id/cancel     # Cancel booking
GET    /bookings/my             # My bookings
```

#### **Payments Service**
```
POST   /payments/create-order   # Create Razorpay order
POST   /payments/verify         # Verify payment
GET    /payments/:id            # Get payment status
POST   /payments/refund         # Request refund
```

#### **Partners Service**
```
POST   /partners/register       # Partner registration
GET    /partners/me             # Partner profile
PATCH  /partners/me             # Update partner profile
GET    /partners/events         # Partner's events
GET    /partners/bookings       # Partner's bookings
GET    /partners/analytics      # Partner analytics
GET    /partners/payouts        # Payout history
POST   /partners/payouts/request# Request payout
```

#### **Notifications Service**
```
POST   /notifications/send      # Send notification (internal)
GET    /notifications           # Get user notifications
PATCH  /notifications/:id/read  # Mark as read
POST   /notifications/preferences # Update preferences
```

#### **Admin Service**
```
GET    /admin/reports           # Get reported content
POST   /admin/reports/:id/action# Take action on report
GET    /admin/users             # List users
POST   /admin/users/:id/suspend # Suspend user
GET    /admin/partners          # List partners
POST   /admin/partners/:id/verify# Verify partner
GET    /admin/analytics         # Platform analytics
```

### Tables

#### **users**
```sql
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  email VARCHAR(255) UNIQUE NOT NULL,
  auth_provider VARCHAR(20) DEFAULT 'email', -- email, google
  name VARCHAR(100),
  avatar_url TEXT,
  neighborhood VARCHAR(100),
  accessibility_prefs JSONB DEFAULT '{}',
  emergency_contact JSONB,
  role VARCHAR(20) DEFAULT 'user', -- user, host, partner, admin
  is_verified BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
```

#### **clubs**
```sql
CREATE TABLE clubs (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name VARCHAR(50) NOT NULL,
  slug VARCHAR(50) UNIQUE NOT NULL,
  description TEXT,
  icon_url TEXT,
  cover_url TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
```

#### **user_clubs**
```sql
CREATE TABLE user_clubs (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  club_id UUID REFERENCES clubs(id) ON DELETE CASCADE,
  joined_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(user_id, club_id)
);
```

#### **user_interests**
```sql
CREATE TABLE user_interests (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  interest VARCHAR(50) NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
```

#### **events**
```sql
CREATE TABLE events (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  club_id UUID REFERENCES clubs(id),
  host_id UUID REFERENCES users(id),
  partner_id UUID REFERENCES partners(id),
  title VARCHAR(200) NOT NULL,
  description TEXT,
  event_type VARCHAR(20) DEFAULT 'free', -- free, paid
  status VARCHAR(20) DEFAULT 'draft', -- draft, published, cancelled, completed
  starts_at TIMESTAMPTZ NOT NULL,
  ends_at TIMESTAMPTZ,
  location_name VARCHAR(200),
  location_address TEXT,
  location_coords POINT,
  accessibility_notes TEXT,
  capacity INT DEFAULT 20,
  price DECIMAL(10,2) DEFAULT 0,
  currency VARCHAR(3) DEFAULT 'INR',
  cancellation_policy TEXT,
  images TEXT[],
  tags TEXT[],
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
```

#### **rsvps**
```sql
CREATE TABLE rsvps (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  event_id UUID REFERENCES events(id) ON DELETE CASCADE,
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  status VARCHAR(20) DEFAULT 'confirmed', -- confirmed, waitlist, cancelled
  checked_in BOOLEAN DEFAULT false,
  checked_in_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(event_id, user_id)
);
```

#### **bookings**
```sql
CREATE TABLE bookings (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  event_id UUID REFERENCES events(id),
  user_id UUID REFERENCES users(id),
  payment_id UUID REFERENCES payments(id),
  status VARCHAR(20) DEFAULT 'pending', -- pending, confirmed, cancelled, refunded
  amount DECIMAL(10,2) NOT NULL,
  platform_fee DECIMAL(10,2),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
```

#### **payments**
```sql
CREATE TABLE payments (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id),
  booking_id UUID,
  razorpay_order_id VARCHAR(100),
  razorpay_payment_id VARCHAR(100),
  amount DECIMAL(10,2) NOT NULL,
  currency VARCHAR(3) DEFAULT 'INR',
  status VARCHAR(20) DEFAULT 'pending', -- pending, completed, failed, refunded
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
```

#### **partners**
```sql
CREATE TABLE partners (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id),
  business_name VARCHAR(200) NOT NULL,
  business_type VARCHAR(50),
  description TEXT,
  logo_url TEXT,
  documents JSONB, -- verification docs
  bank_details JSONB,
  commission_rate DECIMAL(5,2) DEFAULT 10.00,
  is_verified BOOLEAN DEFAULT false,
  verified_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
```

#### **event_chats**
```sql
CREATE TABLE event_chats (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  event_id UUID REFERENCES events(id) ON DELETE CASCADE,
  user_id UUID REFERENCES users(id),
  message TEXT NOT NULL,
  message_type VARCHAR(20) DEFAULT 'text', -- text, voice, image
  is_pinned BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
```

#### **badges**
```sql
CREATE TABLE badges (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name VARCHAR(50) NOT NULL,
  description TEXT,
  icon_url TEXT,
  criteria JSONB -- rules for earning
);

CREATE TABLE user_badges (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  badge_id UUID REFERENCES badges(id),
  earned_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(user_id, badge_id)
);
```

#### **reports**
```sql
CREATE TABLE reports (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  reporter_id UUID REFERENCES users(id),
  reported_user_id UUID REFERENCES users(id),
  reported_event_id UUID REFERENCES events(id),
  reason VARCHAR(100) NOT NULL,
  description TEXT,
  status VARCHAR(20) DEFAULT 'pending', -- pending, reviewed, resolved
  admin_notes TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  resolved_at TIMESTAMPTZ
);
```

#### **notifications**
```sql
CREATE TABLE notifications (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),

  type VARCHAR(50) NOT NULL,
  title VARCHAR(200),
  body TEXT,
  data JSONB,
  is_read BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
```
